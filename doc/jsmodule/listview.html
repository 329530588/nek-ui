var component = new RGUI.Component({
    template: template,
    data: {
        source: [
            {name: '选项1'},
            {name: '选项2'},

        ]
    }
});

禁用某一项，禁用组件

<div class="g-row">
    <div class="g-col g-col-6">
        <listView source={source} />
    </div>
    <div class="g-col g-col-6">
        <listView source={source} disabled />
    </div>
</div>

远程数据

<listView service={@(this.service)} />

var component = new RGUI.Component({
    template: template,
    service: {
        getList: function(params, success) {
            RGUI.ajax.request({
                url: '../data/list.json',
                method: 'get',
                type: 'json',
                data: params,
                success: success
            });
        }
    }
});
