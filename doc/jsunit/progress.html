进度条 Progress

示例

基本形式

<progress percent="36" />

或

var progress = new Progress({
    data: {
        percent: 36
    }
}).$inject('#j-example1');

颜色扩展

<progress percent="25" type="success" />
<progress percent="50" type="warning" />
<progress percent="75" type="error" />

或

var progress = new Progress({
    data: {
        percent: 25,
        type: 'success'
    }
}).$inject('#j-example2');
var progress = new Progress({
    data: {
        percent: 50,
        type: 'warning'
    }
}).$inject('#j-example2');
var progress = new Progress({
    data: {
        percent: 75,
        type: 'error'
    }
}).$inject('#j-example2');

尺寸扩展

<progress percent="25" size="xs" />
<progress percent="50" size="sm" />
<progress percent="75" />

或

var progress = new Progress({
    data: {
        percent: 25,
        text: false,
        size: 'xs'
    }
}).$inject('#j-example3');
var progress = new Progress({
    data: {
        percent: 50,
        text: false,
        size: 'sm'
    }
}).$inject('#j-example3');
var progress = new Progress({
    data: {
        percent: 75,
        text: false,
        size: null
    }
}).$inject('#j-example3');

条纹

<progress percent="36" striped={true} />

或

var progress = new Progress({
    data: {
        percent: 36,
        striped: true
    }
}).$inject('#j-example4');

条纹动画

<progress percent="72" type="error" striped={true} active={true} />

或

var progress = new Progress({
    data: {
        percent: 72,
        type: 'error',
        striped: true,
        active: true
    }
}).$inject('#j-example5');

API

Class

Progress

继承自Component。

Options

new Progress({options})
参数类型默认值描述
dataobject绑定属性
data.percentnumber36百分比
data.textstring | booleantrue在进度条中是否显示百分比。值为`string`时显示该段文字。
data.sizestringnull进度条的尺寸
data.typestringnull进度条的类型，改变显示颜色
data.stripedbooleanfalse是否显示条纹
data.activebooleanfalse进度条是否为激活状态，当`striped`为`true`时，进度条显示动画
